MenuButton(
                                        label: 'Files',
                                        icon: Icons.file_present,
                                        onPressed: () async {
                                          final statusX =
                                              await getMediaPermission();
                                          print("Final Status = $statusX");
                                          if (statusX.isGranted) {
                                            FilePickerResult? pickedFile =
                                                await FilePicker.platform
                                                    .pickFiles(
                                              // allowMultiple: true,
                                              type: FileType.custom,
                                              allowedExtensions: ['jpg', 'png'],
                                            );
                                            if (pickedFile != null) {
                                              for (int i = 0;
                                                  i < pickedFile.files.length;
                                                  i++) {
                                                File selectedFile = File(
                                                    pickedFile.files[i].path
                                                        .toString());
                                                print(selectedFile);
                                              }
                                            } else {
                                              print("error");
                                            }
                                          } else {
                                            print(
                                                "Error for request permission manageExternalStorage");
                                          }
                                        },
                                      ),
